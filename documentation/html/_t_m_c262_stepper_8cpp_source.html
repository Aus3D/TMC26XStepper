00287     unsigned char current_scaling = 0;
00288         //calculate the current scaling from the max current setting (in mA)
00289         double mASetting = (double)current;
00290     double resistor_value = (double) this->resistor;
00291         // remove vesense flag
00292         this->driver_configuration_register_value &= ~(VSENSE); 
00293         //this is derrived from I=(cs+1)/32*(Vsense/Rsense)
00294     //leading to cs = CS = 32*R*I/V (with V = 0,31V oder 0,165V  and I = 1000*current)
00295         //with Rsense=0,15
00296         //for vsense = 0,310V (VSENSE not set)
00297         //or vsense = 0,165V (VSENSE set)

00301         if (current_scaling<16) {
00302         //set the csense bit to get a use half the sense voltage (to support lower motor currents)
00303                 this->driver_configuration_register_value |= VSENSE;
00304         //and recalculate the current setting
00305         current_scaling = (byte)((resistor_value*mASetting*32.0/(0.165*1000.0*1000.0))-0.5); //theoretically - 1.0 for better rounding it is 0.5
00306 //#ifdef DEBUG
00307                 Serial.print("CS (Vsense=1): ");

00310         Serial.print("CS: ");
00311         Serial.println(current_scaling);

00317                 current_scaling=31;

00320         stall_guard2_current_register_value &= ~(CURRENT_SCALING_PATTERN);

00606 /*
00607  * In a constant off time chopper scheme both coil choppers run freely, i.e. are not synchronized. 
00608  * The frequency of each chopper mainly depends on the coil current and the position dependant motor coil inductivity, thus it depends on the microstep position. 
00609  * With some motors a slightly audible beat can occur between the chopper frequencies, especially when they are near to each other. This typically occurs at a 
00610  * few microstep positions within each quarter wave. This effect normally is not audible when compared to mechanical noise generated by ball bearings, etc. 
00611  * Further factors which can cause a similar effect are a poor layout of sense resistor GND connection.
00612  * Hint: A common factor, which can cause motor noise, is a bad PCB layout causing coupling of both sense resistor voltages 
00613  * (please refer to sense resistor layout hint in chapter 8.1).
00614  * In order to minimize the effect of a beat between both chopper frequencies, an internal random generator is provided. 
00615  * It modulates the slow decay time setting when switched on by the RNDTF bit. The RNDTF feature further spreads the chopper spectrum, 
00616  * reducing electromagnetic emission on single frequencies.
00617  */
